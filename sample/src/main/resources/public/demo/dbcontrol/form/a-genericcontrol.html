源码

HTML

    //DBLabel，HTML元素为Label
    <label data-jslet="type: 'DBLabel', dataset: 'employee', field: 'id'"></label>
    
    //DBDataLabel，HTML元素为Label
    <label data-jslet="type: 'DBDataLabel', dataset: 'employee', field: 'id'"></label>
    
    //DBCheckBox，HTML元素为Input
    <input data-jslet="type: 'DBCheckBox', dataset: 'employee', field: 'married'"></input>
    
    //DBPassword，HTML元素为Input
    <input data-jslet="type: 'DBPassword', dataset: 'employee', field: 'name'"></input>
    
    //DBTextArea，HTML元素为TextArea
    <TextArea data-jslet="type: 'DBTextArea', dataset: 'employee', field: 'summary'"></TextArea>
JavaScript

	//datasetMetaStore定义在公共js:common/datasetmetastore.js中
	//将数据集定义信息仓库加到datasetFactory中，创建Dataset时会仓库里去定义信息
	jslet.data.datasetFactory.addMetaStore(window.datasetMetaStore);
	//通过工厂方法，可以自动创建主数据集及相关的数据集
    jslet.data.datasetFactory.createDataset('employee').done(function() {
    	jslet.data.getDataset('employee').query();
    	jslet.ui.install();
    });
